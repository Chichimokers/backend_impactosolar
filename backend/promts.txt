on:
  push:
    branches:
      - master

jobs:
  build-and-deploy:
    runs-on: self-hosted
    steps:
      - name: Clean all __pycache__
        run: |
          sudo find ${{ github.workspace }} -name "__pycache__" -type d -exec rm -rf {} +
      - name: Checkout code
        uses: actions/checkout@v3
    
      - name: Build and run with Docker Compose
        run: |
          docker-compose down
          docker-compose up --build -d
      - name: Run makemigrations for all apps
        run: |
          docker-compose exec  -T web  python manage.py makemigrations core
          docker-compose exec  -T web python manage.py makemigrations products
          docker-compose exec  -T web python manage.py makemigrations sales
          docker-compose exec  -T web python manage.py makemigrations users
          docker-compose exec  -T web python manage.py makemigrations history

          docker-compose exec  -T web  python manage.py migrate core
          docker-compose exec  -T web python manage.py migrate products
          docker-compose exec  -T web python manage.py migrate sales
          docker-compose exec  -T web python manage.py migrate users
          docker-compose exec  -T web python manage.py migrate history
      - name: Run migrate
        run: |
          docker-compose exec -T web python manage.py makemigrations
          docker-compose exec -T web python manage.py migrate

      - name: Run migrate_all_tenants
        run: |
          docker-compose exec -T web python manage.py migrate_all_tenants
